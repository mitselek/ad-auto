# ec-clear — run every Eternity Challenge to 5 completions.
# Stage: once your EP comfortably clears EC trees (typically each Reality).
# Setup: save one tree per challenge as presets named EC1 .. EC12
#        (a single generic EC tree also works — point every load line at it).
# Repeat toggle: OFF (finishes and stops).
# Note: ECs must be completed in a sane order; this runs 1 -> 12.

auto infinity on
auto eternity off

until ec1 completions >= 5 {
  studies nowait load name EC1
  unlock nowait ec1
  start ec1
  wait pending completions >= 1
  eternity nowait
}
